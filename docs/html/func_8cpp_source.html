92void handleMenuState(const sf::Event &event, GameState &state,
93 sf::RenderWindow &window, sf::Text &startButton,
94 sf::Text &settingsButton, sf::Text &saveButton,
95 sf::Text &quitButton) {
96 if (event.type == sf::Event::MouseButtonPressed &&
97 event.mouseButton.button == sf::Mouse::Left) {
98 sf::Vector2i mousePos = sf::Mouse::getPosition(window);
99 if (startButton.getGlobalBounds().contains(mousePos.x, mousePos.y)) {
100 state = GameState::Playing;
101 } else if (settingsButton.getGlobalBounds().contains(mousePos.x,
102 mousePos.y)) {
103 state = GameState::Settings;
104 } else if (saveButton.getGlobalBounds().contains(mousePos.x, mousePos.y)) {
105 state = GameState::SaveLoad;
106 } else if (quitButton.getGlobalBounds().contains(mousePos.x, mousePos.y)) {
107 window.close();
108 }
109 }

119void handlePlayingState(const sf::Event &event, GameState &state,
120 int &currentDialogueIndex1,
121 std::vector<std::string> &dialogues1,
122 sf::Text &dialogue1, bool &isCharacter1Speaking) {
123 if (event.type == sf::Event::KeyPressed) {
124 if (event.key.code == sf::Keyboard::Escape) {
125 state = GameState::Menu;
126 } else if (event.key.code == sf::Keyboard::LControl ||
127 event.key.code == sf::Keyboard::RControl) {
128 skipToChoices(currentDialogueIndex1, dialogues1, state, dialogue1);
129 }
130 } else if (event.type == sf::Event::MouseButtonPressed &&
131 event.mouseButton.button == sf::Mouse::Left) {
132 if (isCharacter1Speaking) {
133 currentDialogueIndex1++;
134 if (currentDialogueIndex1 < dialogues1.size()) {
135 dialogue1.setString(dialogues1[currentDialogueIndex1]);
136 sf::FloatRect textRect1 = dialogue1.getLocalBounds();
137 dialogue1.setOrigin(textRect1.left + textRect1.width / 2.0f,
138 textRect1.top + textRect1.height / 2.0f);
139 dialogue1.setPosition(sf::Vector2f(
140 1280 / 2.0f, 660));
141
142 if (currentDialogueIndex1 == dialogues1.size() - 1) {
143 state = GameState::Choice;
144 }
145 }
146 }
147 isCharacter1Speaking = !isCharacter1Speaking;
148 }
149}

161void handleSettingsState(const sf::Event &event, GameState &state,
162 sf::RenderWindow &window, sf::Music &music,
163 sf::RectangleShape &volumeSlider,
164 sf::RectangleShape &brightnessSlider,
165 sf::RectangleShape &volumeKnob,
166 sf::RectangleShape &brightnessKnob,
167 sf::Sprite &gameBackground) {
168 if (event.type == sf::Event::MouseButtonPressed &&
169 event.mouseButton.button == sf::Mouse::Left) {
170 sf::Vector2i mousePos = sf::Mouse::getPosition(window);
171 if (volumeSlider.getGlobalBounds().contains(mousePos.x, mousePos.y)) {
172 float volume = (mousePos.x - volumeSlider.getPosition().x) / 2.0f;
173 volume = std::max(0.0f, std::min(100.0f, volume));
174 music.setVolume(volume);
175 volumeKnob.setPosition(volumeSlider.getPosition().x + volume * 2,
176 volumeKnob.getPosition().y);
177 } else if (brightnessSlider.getGlobalBounds().contains(mousePos.x,
178 mousePos.y)) {
179 float brightness = (mousePos.x - brightnessSlider.getPosition().x);
180 brightness = std::max(0.0f, std::min(200.0f, brightness));
181 brightnessKnob.setPosition(brightnessSlider.getPosition().x + brightness,
182 brightnessKnob.getPosition().y);
183
184 sf::Color color = gameBackground.getColor();
185 color.r = color.g = color.b = 255 * (brightness / 200.0f);
186 gameBackground.setColor(color);
187 }
188 } else if (event.type == sf::Event::KeyPressed &&
189 event.key.code == sf::Keyboard::Escape) {
190 state = GameState::Menu;
191 }
192}
